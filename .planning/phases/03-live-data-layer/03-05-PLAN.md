      const [bid, setBid]             = uS_est(null);
      const [lineItems, setLineItems] = uS_est(null);  // null=loading, []=empty
      const [libItems, setLibItems]   = uS_est(null);
      const [libQ, setLibQ]           = uS_est('');
      const [tab, setTab]             = uS_est('library');
      const [editingId, setEditingId] = uS_est(null);

      uE_est(() => {
        if (!activeBidId) return;
        window.sb.from('bids').select('*').eq('id', activeBidId).single()
          .then(({ data }) => { if (data) setBid(data); });
      }, [activeBidId]);

      // Load line items
      uE_est(() => {
        if (!activeBidId) { setLineItems([]); return; }
        let cancelled = false;
        async function load() {
          const { data, error } = await window.dbHelpers.getLineItems(activeBidId);
          if (!cancelled) setLineItems(error ? [] : (data || []));
        }
        load();
        return () => { cancelled = true; };
      }, [activeBidId]);

      // Load library items (once)
      uE_est(() => {
        let cancelled = false;
        async function load() {
          const { data } = await window.dbHelpers.getLibraryItems();
          if (!cancelled) setLibItems(data || []);
        }
        load();
        return () => { cancelled = true; };
      }, []);

      const fuse = uM_est(() => {
        if (!libItems || !libItems.length) return null;
        return typeof window.Fuse === 'function'
          ? new window.Fuse(libItems, { keys:['description','code','category'], threshold:0.3 })
          : null;
      }, [libItems]);

      const libResults = uM_est(() => {
        if (!libItems) return [];
        if (!libQ || !fuse) return libItems;
        return fuse.search(libQ).map(r => r.item);
      }, [libQ, fuse, libItems]);

      const total = uM_est(() => {
        if (!lineItems) return 0;
        return lineItems.reduce((sum, r) => sum + (r.total != null ? Number(r.total) : (r.qty||0)*(r.unit_cost||0)), 0);
      }, [lineItems]);

      const oh = total * 0.08;
      const profit = total * 0.105;
      const bond = total * 0.012;
      const grand = total + oh + profit + bond;

      async function handleAddLine() {
        if (!activeBidId) return;
        const { data: newRow, error } = await window.dbHelpers.addLineItem({
          bid_id: activeBidId, description: 'New item', qty: 1, unit: 'LS', unit_cost: 0,
          sort_order: (lineItems || []).length,
        });
        if (!error && newRow) setLineItems(prev => [...(prev || []), newRow]);
      }

      async function handleUpdate(id, field, value) {
        const parsed = field === 'qty' || field === 'unit_cost' ? parseFloat(value) || 0 : value;
        setLineItems(prev => prev.map(r => r.id === id ? { ...r, [field]: parsed } : r));
        await window.dbHelpers.updateLineItem(id, { [field]: parsed });
      }

      async function handleDelete(id) {
        if (!confirm('Delete this line item? This cannot be undone.')) return;
        const { error } = await window.dbHelpers.deleteLineItem(id);
        if (!error) setLineItems(prev => prev.filter(r => r.id !== id));
      }

      async function handleInsertFromLibrary(lib) {
        if (!activeBidId) return;
        const unit_cost = (lib.material_cost || 0) + (lib.labor_cost || 0);
        const { data: newRow, error } = await window.dbHelpers.addLineItem({
          bid_id: activeBidId, description: lib.description, qty: 1,
          unit: lib.unit || 'EA', unit_cost,
          sort_order: (lineItems || []).length,
        });
        if (!error && newRow) setLineItems(prev => [...(prev || []), newRow]);
      }

      // No activeBidId — prompt user to select a bid
      if (!activeBidId) {
        return (
          <div className="view active" style={{display:'flex',alignItems:'center',justifyContent:'center',height:'60vh'}}>
            <window.EmptyState
              heading="No line items"
              body="Select a bid from the Pipeline to open its estimate."
            />
          </div>
        );
      }

      if (lineItems === null) return <window.Spinner />;

      return (
        <div className="view active">
          {/* Bid header */}
          <div style={{padding:'10px 24px',borderBottom:'1px solid var(--line)',background:'var(--paper-2)',display:'flex',alignItems:'center',gap:14}}>
            <div>
              <div style={{fontSize:13,fontWeight:700}}>{bid ? bid.name : '…'}</div>
              <div className="muted" style={{fontSize:11.5}}>
                {bid?.gc_name && <span>{bid.gc_name} · </span>}
                {bid?.due_date && <span>Due <b style={{color:'var(--accent)'}}>{bid.due_date}</b></span>}
              </div>
            </div>
            <div className="bid-totals" style={{marginLeft:'auto'}}>
              <div className="t"><span className="l">Direct cost</span><span className="v">{fmt(total)}</span></div>
              <div className="divider"></div>
              <div className="t"><span className="l">OH + Profit</span><span className="v">{fmt(oh+profit)}</span></div>
              <div className="divider"></div>
              <div className="t main"><span className="l">Bid total</span><span className="v">{fmt(grand)}</span></div>
            </div>
          </div>

          {/* 3-pane layout */}
          <div style={{display:'grid',gridTemplateColumns:'1fr 320px',height:'calc(100vh - 52px - 68px)',overflow:'hidden'}}>

            {/* Line items grid */}
            <div style={{overflow:'auto'}}>
              <div style={{padding:'8px 14px',borderBottom:'1px solid var(--line)',display:'flex',alignItems:'center',gap:8,background:'var(--paper)',position:'sticky',top:0,zIndex:5}}>
                <button className="btn sm" onClick={handleAddLine}><Icon.plus/> Add line</button>
                <div className="spacer"></div>
                <span className="muted" style={{fontSize:11.5}}>{lineItems.length} lines</span>
              </div>

              {lineItems.length === 0 ? (
                <window.EmptyState
                  heading="No line items"
                  body="This estimate is empty. Add line items to build the scope."
                  action={{ label:'Add Line Item', onClick:handleAddLine }}
                />
              ) : (
                <table className="wf">
                  <thead>
                    <tr>
                      <th style={{paddingLeft:16}}>Description</th>
                      <th style={{width:60}} className="ctr">UOM</th>
                      <th style={{width:90}} className="num">Qty</th>
                      <th style={{width:100}} className="num">Unit $</th>
                      <th style={{width:120}} className="num">Extended</th>
                      <th style={{width:30}}></th>
                    </tr>
                  </thead>
                  <tbody>
                    {lineItems.map((r) => {
                      const ext = (r.qty||0) * (r.unit_cost||0);
                      return (
                        <tr key={r.id} style={{position:'relative'}}
                          onMouseEnter={e => e.currentTarget.querySelector('.del-btn')?.style && (e.currentTarget.querySelector('.del-btn').style.opacity='1')}
                          onMouseLeave={e => e.currentTarget.querySelector('.del-btn')?.style && (e.currentTarget.querySelector('.del-btn').style.opacity='0')}
                        >
                          <td style={{paddingLeft:16}}>
                            <input className="inline-inp" defaultValue={r.description}

                              onKeyDown={e => e.key==='Enter' && e.target.blur()}
                            />
                          </td>
                          <td className="ctr">
                            <select className="chip mono" value={r.unit||'LS'} onChange={e => handleUpdate(r.id,'unit',e.target.value)}
                              style={{background:'transparent',border:'none',fontSize:11,fontFamily:'var(--mono)'}}>
                              {['EA','LF','SF','SY','CY','LS','HR','TON'].map(u=><option key={u}>{u}</option>)}
                            </select>
                          </td>
                          <td className="num">
                            <input className="inline-inp num tnum" defaultValue={r.qty}
                              onBlur={e => handleUpdate(r.id,'qty',e.target.value)}
                              onKeyDown={e => e.key==='Enter' && e.target.blur()}
                            />
                          </td>
                          <td className="num">
                            <input className="inline-inp num tnum" defaultValue={r.unit_cost}
                              onBlur={e => handleUpdate(r.id,'unit_cost',e.target.value)}
                              onKeyDown={e => e.key==='Enter' && e.target.blur()}
                            />
                          </td>
                          <td className="num tnum" style={{fontWeight:600}}>{fmt(ext)}</td>
                          <td className="ctr">
                            <button className="del-btn btn ghost xs" style={{opacity:0,color:'var(--bad)',transition:'opacity .15s'}}
                              onClick={() => handleDelete(r.id)}>✕</button>
                          </td>
                        </tr>
                      );
                    })}
                    <tr className="total-row">
                      <td colSpan={4} style={{paddingLeft:16,textTransform:'uppercase',fontSize:11,letterSpacing:'.08em'}}>Direct cost subtotal</td>
                      <td className="num tnum" style={{fontSize:14}}>{fmt(total)}</td>
                      <td></td>
                    </tr>
                    <tr><td colSpan={4} style={{paddingLeft:16,color:'var(--ink-3)'}}>Overhead (8.0%)</td><td className="num tnum">{fmt(oh)}</td><td></td></tr>
                    <tr><td colSpan={4} style={{paddingLeft:16,color:'var(--ink-3)'}}>Profit (10.5%)</td><td className="num tnum">{fmt(profit)}</td><td></td></tr>

                      <td colSpan={4} style={{paddingLeft:16,fontSize:12}}>BID TOTAL</td>
                      <td className="num tnum" style={{fontSize:16,color:'var(--accent)'}}>{fmt(grand)}</td>
                      <td></td>
                    </tr>
                  </tbody>
                </table>
              )}
            </div>

            {/* Side panel — library tab */}
            <aside className="side">
              <div className="tabs">
                <button className={tab==='library'?'active':''} onClick={()=>setTab('library')}>Library</button>
                <button className={tab==='findings'?'active':''} onClick={()=>setTab('findings')}>Findings</button>
              </div>
              <div className="pane">
                {tab==='library' && (
                  <>
                    <div style={{padding:'8px 10px 4px'}}>
                      <input value={libQ} onChange={e=>setLibQ(e.target.value)} placeholder="Search library…"
                        style={{width:'100%',border:'1px solid var(--line)',borderRadius:'var(--r-sm)',padding:'5px 8px',fontSize:12,background:'var(--paper)'}}/>
