# Plan 02-03: Auth Gate Integration

## Objective

Wire the auth gate into `shell.jsx` using `onAuthStateChange` so the app shows `LoginView` when logged out and the full Shell when logged in. Add logout to the Topbar avatar. Display the real signed-in user's initials in the avatar. All changes are in `project/shell.jsx` — there is no `app.jsx`.

## must_haves

```yaml
truths:
  - App() in shell.jsx has a session state initialized to undefined (not null — prevents flash)
  - App() subscribes to window.sb.auth.onAuthStateChange in a useEffect with cleanup
  - App() returns null when session === undefined (loading — no flash of content)
  - App() renders <LoginView /> when session is null (logged out)
  - App() renders <AuthenticatedApp session={session} /> when session is truthy (hooks-safe split)
  - AuthenticatedApp derives user initials from session.user.email and passes them to Topbar
  - Topbar accepts initials prop and renders them instead of hardcoded EP
  - Topbar accepts onLogout prop; avatar click shows Sign out option
  - Clicking Sign out calls window.sb.auth.signOut() which triggers onAuthStateChange → session null
  - Session persists across page refresh (Supabase v2 auto-restores from localStorage via INITIAL_SESSION event)
goal: "Auth gate works end-to-end: unauthenticated → login screen, authenticated → app with real initials, logout → login screen, refresh → stays logged in"
```

<threat_model>
## Threat Model (ASVS L1)

| Threat | Severity | Mitigation |
|--------|----------|------------|
| Flash of authenticated content before session check | MEDIUM | `session === undefined` initial state returns null (blank) until INITIAL_SESSION fires — no flash |
| Memory leak from unsubscribed auth listener | LOW | useEffect cleanup calls `subscription.unsubscribe()` |
| Logout without server-side token invalidation | MEDIUM | `window.sb.auth.signOut()` invalidates the token server-side and clears localStorage |
| Navigation bypass via localStorage view key | LOW | Auth gate is in App() root — even if localStorage stores a view ID, App renders LoginView when session is null |
| Email in avatar title attribute leaks PII in screenshots | LOW | Avatar shows initials only; title attribute set to initials, not full email |
</threat_model>

```jsx
function AuthenticatedApp({ session }) {
  const [active, setActive] = useState(() => {
    try { return localStorage.getItem('fs-view') || 'home'; } catch { return 'home'; }
  });
  const go = (id) => { setActive(id); try { localStorage.setItem('fs-view', id); } catch {} };
  useEffect(() => { window.__go = go; }, []);

  // Derive initials from email (e.g. "evan.pruitt@fs.com" → "EP")
  const initials = useMemo(() => {
    const email = session?.user?.email || '';
    const parts = email.split('@')[0].split(/[._-]/);
    return parts.slice(0, 2).map(p => p[0]?.toUpperCase() || '').join('') || 'U';
  }, [session]);

  const crumb = useMemo(() => {
    switch (active) {
      case 'home': return <><b>Home</b> <span className="sep">/</span> <span>Pipeline overview</span></>;
      case 'pipeline': return <><b>Pipeline</b> <span className="sep">/</span> <span>All stages</span></>;
      case 'estimator': return <><span>Bids</span> <span className="sep">/</span> <b>Denver Regional Lab</b> <span className="sep">/</span> <span>v3 draft</span></>;
      case 'jobs': return <><b>Jobs</b> <span className="sep">/</span> <span>Active · 2026</span></>;
      case 'job': return <><span>Jobs</span> <span className="sep">/</span> <b>Ada Co. Courthouse · 26-040</b></>;
      case 'co': return <><span>Jobs</span> <span className="sep">/</span> <span>Ada Co. Courthouse</span> <span className="sep">/</span> <b>Change Orders</b></>;
      case 'bidhistory': return <><span>Insight</span> <span className="sep">/</span> <b>Bid History &amp; Comparison</b></>;
      default: return <b>{NAV.find(n=>n.id===active)?.label || 'Home'}</b>;
    }
  }, [active]);

  const actions = useMemo(() => {
    switch (active) {
      case 'home':
      case 'pipeline':
        return <>
          <button className="btn ghost" onClick={()=>go('pipeline')}>Ingest ITB</button>
          <button className="btn primary" onClick={()=>go('estimator')}><Icon.plus /> New bid</button>
        </>;
      case 'estimator':
        return <>
          <button className="btn ghost">History</button>
          <button className="btn">Compare</button>
          <button className="btn accent">Build proposal →</button>
        </>;
      case 'jobs':
        return <>
          <button className="btn ghost">Import from awarded</button>
          <button className="btn primary"><Icon.plus /> New job</button>
        </>;
      case 'job':
        return <>
          <button className="btn ghost">Open files</button>
          <button className="btn">Edit scope</button>
          <button className="btn accent">+ Change order</button>
        </>;
      case 'co':
        return <>
          <button className="btn ghost" onClick={()=>go('job')}><Icon.back /> Job</button>
          <button className="btn accent">+ New CO</button>
        </>;
      default: return null;
    }
  }, [active]);

  const View = window.Views?.[active] || window.Views?.home;

  async function handleLogout() {
    await window.sb.auth.signOut();
  }

  return (
    <div id="app">
      <Topbar crumb={crumb} actions={actions} initials={initials} onLogout={handleLogout} />
      <div id="main">
        <Rail active={active} onGo={go} />
        <main id="content">
          {View ? <View go={go} /> : <div style={{padding:40}}>Loading…</div>}
        </main>
      </div>
    </div>
  );
}

function App() {
  const [session, setSession] = useState(undefined);

  useEffect(() => {
    const { data: { subscription } } = window.sb.auth.onAuthStateChange((_event, s) => {
      setSession(s);
    });
    return () => subscription.unsubscribe();
  }, []);

  if (session === undefined) return null;
  if (!session) return <LoginView />;
  return <AuthenticatedApp session={session} />;
}
```
</action>

```jsx
function Topbar({ crumb, actions, initials, onLogout }) {
  const [showMenu, setShowMenu] = useState(false);
  return (
    <header id="topbar">
      <div className="brand">
        <div className="monogram">F&amp;S</div>
        <div className="name">Estimating Suite <em>v1</em></div>
      </div>
      <span className="sep">/</span>
      <div className="crumb">{crumb}</div>
      <div className="spacer"></div>
      <div className="search">
        <Icon.search /> <span>Search bids, jobs, contacts, items…</span>
        <span className="kbd">⌘K</span>
      </div>
      {actions}
      <div className="avatar-wrap">
        <div className="avatar" title={initials} onClick={() => setShowMenu(v => !v)}>
          {initials}
        </div>
        {showMenu && (
          <div className="avatar-menu">
            <button className="avatar-menu-item" onClick={() => { setShowMenu(false); onLogout(); }}>
              Sign out
            </button>
          </div>
        )}
      </div>
    </header>
  );
}
```
